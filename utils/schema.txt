CREATE TABLE users (
  id INT AUTO_INCREMENT PRIMARY KEY,
  name VARCHAR(100),
  email VARCHAR(100) NOT NULL UNIQUE,
  password VARCHAR(255) NOT NULL,
  role VARCHAR(20) DEFAULT 'user', -- 'user' or 'page'
  website_url VARCHAR(255), -- optional, mostly for pages
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
 
CREATE TABLE user_profiles (
  id INT AUTO_INCREMENT PRIMARY KEY,
  user_id INT NOT NULL,
  bio TEXT,
  profile_pic_url VARCHAR(255),
  FOREIGN KEY (user_id) REFERENCES users(id)
);

CREATE TABLE layers (
  id INT AUTO_INCREMENT PRIMARY KEY,
  name VARCHAR(100) NOT NULL UNIQUE
);

CREATE TABLE post_category_templates (
  id INT AUTO_INCREMENT PRIMARY KEY,
  name VARCHAR(100) NOT NULL UNIQUE,
  shape VARCHAR(30) NOT NULL,
  icon_name VARCHAR(50) NOT NULL,
  color VARCHAR(20),
  class_name VARCHAR(100)
);

CREATE TABLE category_layer_map (
  id INT AUTO_INCREMENT PRIMARY KEY,
  category_id INT NOT NULL,
  layer_id INT NOT NULL,
  FOREIGN KEY (category_id) REFERENCES post_category_templates(id),
  FOREIGN KEY (layer_id) REFERENCES layers(id)
);

CREATE TABLE user_posts (
  id INT AUTO_INCREMENT PRIMARY KEY,
  page_id INT NOT NULL,
  title VARCHAR(255),
  description TEXT,
  image_url TEXT,
  latitude DECIMAL(10, 7),
  longitude DECIMAL(10, 7),
  category_id INT NOT NULL,
  delete_after_hours INT,
  is_permanent BOOLEAN NOT NULL DEFAULT FALSE,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (page_id) REFERENCES users(id),
  FOREIGN KEY (category_id) REFERENCES post_category_templates(id)
);

CREATE TABLE post_layer_map (
  id INT AUTO_INCREMENT PRIMARY KEY,
  post_id INT NOT NULL,
  layer_id INT NOT NULL,
  FOREIGN KEY (post_id) REFERENCES user_posts(id),
  FOREIGN KEY (layer_id) REFERENCES layers(id)
);

CREATE TABLE languages (
  id INT AUTO_INCREMENT PRIMARY KEY,
  name VARCHAR(100) NOT NULL,
  code VARCHAR(10) NOT NULL
);

CREATE TABLE user_news_details (
  id INT AUTO_INCREMENT PRIMARY KEY,
  post_id INT NOT NULL UNIQUE,
  article_url TEXT,
  summary TEXT,
  language_id INT,
  is_high_priority BOOLEAN DEFAULT FALSE,
  is_breaking BOOLEAN DEFAULT FALSE,
  breaking_expire_at TIMESTAMP,
  FOREIGN KEY (post_id) REFERENCES user_posts(id),
  FOREIGN KEY (language_id) REFERENCES languages(id)
);

CREATE TABLE user_post_registrations (
  id INT AUTO_INCREMENT PRIMARY KEY,

  post_id INT NOT NULL,
  user_id INT NOT NULL,

  user_latitude DECIMAL(10, 7),
  user_longitude DECIMAL(10, 7),

  note TEXT,
  resume_url TEXT,

  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

  UNIQUE KEY unique_registration (post_id, user_id),

  FOREIGN KEY (post_id) REFERENCES user_posts(id),
  FOREIGN KEY (user_id) REFERENCES users(id)
);

CREATE TABLE user_post_likes (
  user_id INT NOT NULL,
  post_id INT NOT NULL,
  liked_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (user_id, post_id),
  FOREIGN KEY (user_id) REFERENCES users(id),
  FOREIGN KEY (post_id) REFERENCES user_posts(id)
);

CREATE TABLE user_job_details (
  id INT AUTO_INCREMENT PRIMARY KEY,
  post_id INT NOT NULL UNIQUE,
  job_type ENUM('Internship', 'Jobs', 'Gigs', 'Others') NOT NULL,
  link VARCHAR(500),
  is_paid BOOLEAN NOT NULL DEFAULT TRUE,
  salary_or_stipend VARCHAR(100),
  location_type ENUM('remote', 'onsite', 'hybrid') NOT NULL,
  duration VARCHAR(100),
  application_deadline DATE,
  event_name VARCHAR(255),
  event_date DATE,
  additional_info TEXT,
  FOREIGN KEY (post_id) REFERENCES user_posts(id)
);

CREATE INDEX idx_user_job_details_post_id ON user_job_details(post_id);
CREATE INDEX idx_user_job_details_job_type ON user_job_details(job_type);
CREATE INDEX idx_user_job_details_location_type ON user_job_details(location_type);
CREATE INDEX idx_user_job_details_application_deadline ON user_job_details(application_deadline);

CREATE TABLE skills (
  id INT AUTO_INCREMENT PRIMARY KEY,
  name VARCHAR(100) UNIQUE NOT NULL
);

CREATE TABLE job_skills_map (
  id INT AUTO_INCREMENT PRIMARY KEY,
  job_id INT NOT NULL,
  skill_id INT NOT NULL,
  FOREIGN KEY (job_id) REFERENCES user_job_details(id),
  FOREIGN KEY (skill_id) REFERENCES skills(id)
);

CREATE TABLE education_qualifications (
  id INT AUTO_INCREMENT PRIMARY KEY,
  name VARCHAR(100) UNIQUE NOT NULL
);

CREATE TABLE job_education_map (
  id INT AUTO_INCREMENT PRIMARY KEY,
  job_id INT NOT NULL,
  education_id INT NOT NULL,
  FOREIGN KEY (job_id) REFERENCES user_job_details(id),
  FOREIGN KEY (education_id) REFERENCES education_qualifications(id)
);

CREATE TABLE job_experience (
  id INT AUTO_INCREMENT PRIMARY KEY,
  job_id INT NOT NULL,
  min_years INT DEFAULT 0,
  max_years INT DEFAULT 0,
  FOREIGN KEY (job_id) REFERENCES user_job_details(id)
);

CREATE TABLE `user_followed_pages` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `user_id` INT NOT NULL,
  `page_id` INT NOT NULL,
  `followed_at` TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  INDEX `idx_user_page` (`user_id`, `page_id`),
  FOREIGN KEY (`user_id`) REFERENCES `users`(`id`) ON DELETE CASCADE,
  FOREIGN KEY (`page_id`) REFERENCES `users`(`id`) ON DELETE CASCADE
);

CREATE TABLE `user_followed_layers` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `user_id` INT NOT NULL,
  `layer_id` INT NOT NULL,
  `followed_at` TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  INDEX `idx_user_layer` (`user_id`, `layer_id`),
  FOREIGN KEY (`user_id`) REFERENCES `users`(`id`) ON DELETE CASCADE,
  FOREIGN KEY (`layer_id`) REFERENCES `layers`(`id`) ON DELETE CASCADE
);
